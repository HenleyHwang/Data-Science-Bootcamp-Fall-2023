1. Pull total number of orders that were completed on 18th March 2023

  SELECT COUNT(distinct order_id) AS orders 
  FROM sales 
  WHERE date = “03-18-2023” 

2. Pull total number of orders that were completed on 18th March 2023 with the first name ‘John’ and last name ‘Doe’

  SELECT COUNT(distinct sales.order_id) AS orders
  FROM sales 
    LEFT JOIN customers 
    ON sales.Customer_id = customers.Customer_id 
  WHERE sales.date = “03-18-2023” 
    AND customers.First_name = "John" 
    AND customers.Last_name = "Doe" 

3. Pull total number of customers that purchased in January 2023 and the average amount spend per customer
  
  SELECT SUM(revenue) / COUNT(distinct Customer_id) AS revenue_per_customer
  FROM sales 
  WHERE CAST(date AS string) 
  LIKE “01-%-2023”

4. Pull the departments that generated less than $600 in 2022

  SELECT items.department AS dept, SUM(sales.revenue) AS rev
  FROM sales 
    LEFT JOIN items
    ON sales.Item_id = items.Item_id
  WHERE CAST(sales.date AS string) 
  LIKE “%-%-2022”
  GROUP BY dept
  HAVING rev < 600

5. What is the most and least revenue we have generated by an order

  SELECT MIN(rev), MAX(rev)
  FROM (
    SELECT order_id, SUM(revenue) AS rev
    FROM sales
    GROUP BY order_id
  )

6. What were the orders that were purchased in our most lucrative order

  SELECT order_id, SUM(revenue) AS rev
  FROM sales
  GROUP BY order_id
  ORDER BY rev DESC
